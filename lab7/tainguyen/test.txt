### CATEGORY

# GET all categories
http://localhost:8080/api/categories

# GET category by id
http://localhost:8080/api/categories/1

# POST create new category
http://localhost:8080/api/categories
{
  "id": 6,
  "name": "Tai nghe"
}

# PUT update category
http://localhost:8080/api/categories/6
{
  "id": 6,
  "name": "Tai nghe (update)"
}

# DELETE category
http://localhost:8080/api/categories/6
